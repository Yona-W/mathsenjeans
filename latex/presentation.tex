\section{Notre programme}
\subsection{Principe}
\begin{frame}
\frametitle{Explication}
\begin{itemize}
\onslide<2-5>\item Le programme cherche les meilleurs coups.
\onslide<3-5>\item À chaque coup possible est associé une valeur.
\onslide<4-5>\item Ensuite le coup avec la meilleure valeur est joué.
\end{itemize}
\onslide<5->\center{Comment cette valeur est-elle calculée?}
\end{frame}

\subsection{L'Intelligence}

\begin{frame}
\frametitle{Nos critères d'implémentation}
\framesubtitle{L'avancement du jeu}
\begin{itemize}
\onslide<2->\item Déterminé en grande partie par le nombre de boules en jeu.
\onslide<3->\item La présence d'un grand tas (supérieur à 12).
\onslide<4->\item Permet la différenciation entre les types de stratégies.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Nos critères d'implémentation}
\framesubtitle{L'état du terrain}
\begin{itemize}
\onslide<2->\item "Fonctions d'évaluation".
\onslide<3->\item But: Déterminer la "qualité" d'un état du terrain.
\onslide<4->\item Une fonction par critère (positif ou négatif).
\end{itemize}
\end{frame}

\subsection{L'algorithme résumé}

\begin{frame}
\frametitle{Le fonctionnement}
\onslide<2-> Pour chaque coup possible:
\onslide<3-> \\Essayer de jouer ce coup
\onslide<4-> \\Évaluer son efficacité
\onslide<5-> \\Ensuite, pour le meilleur des coups:
\onslide<6-> \\Si ce coup est légal, jouer ce coup puis attendre le joueur.
\end{frame}

\section{Améliorations possibles}
\subsection{La méthode de Monte Carlo}
\begin{frame}

\frametitle{La méthode de Monte Carlo}
\framesubtitle{Une approche probabiliste}

\begin{itemize}
\onslide<1-> \item Notre "Intelligence Artificielle" ne peut pas prévoir sur plusieurs tours.
\bigskip
\onslide<2-> \large{\item[$\rightarrow$] Comment résoudre cela?}
\onslide<3-> \item Simuler un grand nombre de parties.
\end{itemize}

\end{frame}
\begin{frame}
\frametitle{Explication de l'algorithme}

\onslide<2-> Pour chaque coup possible:
\onslide<3-> \\Essayer de jouer ce coup
\onslide<4-> \\Simuler un grand nombre de parties aléatoires á partir de ce coup
\onslide<5-> \\Enregistrer la fréquence de parties gagnées selon le coup joué
\onslide<6-> \\Jouer le coup avec la plus forte probabilité de gagner

\end{frame}
